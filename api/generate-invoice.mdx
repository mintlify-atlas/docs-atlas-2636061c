## Function Signature

```typescript
function generateInvoice(invoiceData: InvoiceInput): {
  invoice: Invoice;
  accessKey: string;
}
```

## Description

Generates a complete invoice object from the provided invoice data. This function automatically generates an access key (clave de acceso) based on the invoice information and returns both the complete invoice structure and the generated access key.

The access key is generated using the date, document code, RUC, environment, establishment, emission point, and sequential number from the invoice data.

## Return Value

<ResponseField name="invoice" type="Invoice">
  The complete invoice object ready for XML generation.
  
  <Expandable title="Invoice structure">
    <ResponseField name="factura" type="object">
      The root invoice object.
      
      <Expandable title="Properties">
        <ResponseField name="@xmlns:ds" type="string">
          XML namespace for digital signature. Value: `http://www.w3.org/2000/09/xmldsig#`
        </ResponseField>
        
        <ResponseField name="@xmlns:xsi" type="string">
          XML namespace for schema instance. Value: `http://www.w3.org/2001/XMLSchema-instance`
        </ResponseField>
        
        <ResponseField name="@id" type="string">
          Document ID. Value: `comprobante`
        </ResponseField>
        
        <ResponseField name="@version" type="string">
          Document version. Value: `1.0.0`
        </ResponseField>
        
        <ResponseField name="infoTributaria" type="TaxInfo">
          Tax information with the generated access key included.
        </ResponseField>
        
        <ResponseField name="infoFactura" type="InvoiceInfo">
          Invoice information.
        </ResponseField>
        
        <ResponseField name="detalles" type="Details">
          Invoice line items.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="accessKey" type="string">
  The automatically generated access key (clave de acceso) for the invoice. This is a 49-digit numerical string generated based on the invoice information.
</ResponseField>
